  - 物攻: +10
  - 魔攻: +8
  - 物防: +5
  - 魔防: +4

fun realmBonusMult(level: Int): Double {

#### 3.4 等效年份算法
```kotlin
fun calcEffectiveYears(yearOrdinal: Int, qualityOrdinal: Int, percentage: Int): Int {
    val yearBase = when (yearOrdinal) {
        0 -> 100; 1 -> 1000; 2 -> 10000; 3 -> 100000; 4 -> 1000000
        else -> 100
    }
    val yearSpan = yearBase * 9  // 百年的跨度是900, 千年9000...
    val qualityOffset = qualityOrdinal * (yearSpan / 5)  // 每品质占20%跨度
    val pctOffset = percentage * (yearSpan / 5) / 100    // 年分数占剩余20%
    return (yearBase + qualityOffset + pctOffset).coerceAtLeast(1)
}
```

**示例**：
- 千年·精良(72%) = 1000 + 2×1800 + 72×1800/100 = **5,896年**
- 万年·完美(100%) = 10000 + 4×18000 + 100×18000/100 = **100,000年**

fun effectiveQuality(tier: Int): Double {

// 技能倍率：基于有效品质（完美是劣等的9倍）

#### 3.6 魂环负荷系统
```kotlin
fun calcRingLoad(yearOrdinal: Int, qualityOrdinal: Int, percentage: Int): Long {
    val yearLoadBase = longArrayOf(100, 500, 2500, 12500, 62500)
    val qualityMult = doubleArrayOf(1.0, 1.2, 1.5, 1.8, 2.2)
    val pctMult = 1.0 + percentage * 0.003  // 0%=1.0, 100%=1.3
    return (yearLoadBase[yearOrdinal] * qualityMult[qualityOrdinal] * pctMult).toLong()
}
```

**负荷值范围**：

#### 3.7 根骨与吸收容量
```kotlin
// 根骨计算（基于综合属性，不含转生加成）
fun calcRootBone(maxHp: Long, atk: Int, matk: Int, pdef: Int, mdef: Int): Double {
    return atk * 3.0 + matk * 3.0 + pdef * 2.0 + mdef * 2.0 + maxHp / 100.0
}

// 吸收容量 = 根骨 × 6（最低100）
fun calcAbsorptionCapacity(rootBone: Double): Long {
    return (rootBone * 6).toLong().coerceAtLeast(100)
}
```

**示例**：
- Lv.50玩家，atk=500, matk=400, pdef=250, mdef=200, maxHp=5000
- 根骨 = 500×3 + 400×3 + 250×2 + 200×2 + 5000/100 = **4,050**
- 吸收容量 = 4,050 × 6 = **24,300**
- 可吸收：最多2个十万年魂环（12,500×2 = 25,000 > 24,300，超限）

fun cost(combinedTier: Int, ownedCount: Int): Long {
    val year = RingYear.fromCombinedTier(combinedTier)

    return (year.costBase * (1.0 + quality * 0.5) * (1 + ownedCount * 0.3)).toLong()
}
